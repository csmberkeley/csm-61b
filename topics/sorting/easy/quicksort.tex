\question Run the quicksort algorithm. Assume we pick the middle element as the pivot; if there is no exact middle, pick the element to the right of the middle.

\begin{verbatim}
{ 1, 3, 8, 2, 6, 4, 5, 9 }
\end{verbatim}

\begin{solution}[1in]
\begin{verbatim}
{ 1, 3, 2, 4, 5 }, { 6 }, { 8, 9 }
{ 1 }, { 2 }, { 3, 4, 5 }, { 6 }, { 8 }, { 9 }
{ 1 }, { 2 }, { 3 }, { 4 }, { 5 }, { 6 }, { 8 }, { 9 }
\end{verbatim}
\end{solution}
