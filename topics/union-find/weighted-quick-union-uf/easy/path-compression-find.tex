\question Suppose we have the \texttt{WeightedQuickUnionUF} tree with path compression below. Draw the tree after we call: 

\begin{lstlisting} 
find(8);
\end{lstlisting}


\begin{center}
\begin{tikzpicture}[very thick,level/.style={sibling distance=15mm/#1},level distance=30pt]
\node[vertex]{$1$}
child {
    node[vertex]{$2$}
}
child {
    node[vertex]{$3$}
    child {
        node[vertex]{$5$}
    }
}
child {
    node[vertex]{$4$}
    child {
        node[vertex]{$6$}
    }
    child {
        node[vertex]{$7$}
        child {
            node[vertex]{$8$}
        }
    }
};
\end{tikzpicture}
\end{center}




\begin{solution}[1in]
\begin{center}
\begin{tikzpicture}[very thick,level/.style={sibling distance=15mm/#1},level distance=30pt]
\node[vertex]{$1$}
child {
    node[vertex]{$2$}
}
child {
    node[vertex]{$3$}
    child {
        node[vertex]{$5$}
    }
}
child {
    node[vertex]{$4$}
    child {
        node[vertex]{$6$}
    }
}
child {
    node[vertex]{$7$}
}
child {
    node[vertex]{$8$}
};
\end{tikzpicture}
\end{center}

find(8) attempts to find the parent of 8 by searching up the tree through 7, 4, and 1, returning 1. Path compression reassigns the immediate parent of 8, 7, and 4 to 1 (note: that 4's immediate parent doesn't change).
\end{solution}
