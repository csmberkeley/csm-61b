\begin{blocksection}
\question Give the ordering for each depth-first traversal of the tree.

\begin{parts}
\part Pre-order
\begin{solution}[0.25in]
$1 - 2 - 5 - 9 - 7 - 3 - 4$
\end{solution}

\part In-order
\begin{solution}[0.25in]
$5 - 2 - 9 - 1 - 7 - 4 - 3$
\end{solution}

\part Post-order
\begin{solution}[0.25in]
$5 - 9 - 2 - 4 - 3 - 7 - 1$
\end{solution}
\end{parts}
\end{blocksection}
