\begin{blocksection}
\question For each of the following binary trees, determine if the height of
the tree is the same as the height of the optimal tree with the same elements.

\begin{center}
\begin{tikzpicture}[very thick,level/.style={sibling distance=20mm/#1},level distance=30pt]
\node[vertex]{$6$}
child {
    node[vertex]{$5$}
    child {
        node[vertex]{$2$}
        child { edge from parent[draw=none] }
        child {
            node[vertex]{$3$}
            child { edge from parent[draw=none] }
            child {
                node[vertex]{$4$}
            }
        }
    }
    child { edge from parent[draw=none] }
}
child {
    node[vertex]{$9$}
    child {
        node[vertex]{$7$}
    }
    child {
        node[vertex]{$10$}
    }
};
\end{tikzpicture}
\quad
\begin{tikzpicture}[very thick,level/.style={sibling distance=20mm/#1},level distance=30pt]
\node[vertex]{$10$}
child {
    node[vertex]{$6$}
    child {
        node[vertex]{$3$}
        child[sibling distance=10mm] {
            node[vertex]{$1$}
        }
        child[sibling distance=10mm] {
            node[vertex]{$7$}
        }
    }
    child { edge from parent[draw=none] }
}
child {
    node[vertex]{$12$}
    child {
        node[vertex]{$11$}
    }
    child { edge from parent[draw=none] }
};
\end{tikzpicture}
\quad
\begin{tikzpicture}[very thick,level/.style={sibling distance=20mm/#1},level distance=30pt]
\node[vertex]{$9$}
child {
    node[vertex]{$7$}
    child {
        node[vertex]{$5$}
        child[sibling distance=10mm] {
            node[vertex]{$3$}
        }
        child[sibling distance=10mm] {
            node[vertex]{$8$}
        }
    }
    child {
        node[vertex]{$8$}
    }
}
child {
    node[vertex]{$11$}
    child { edge from parent[draw=none] }
    child {
        node[vertex]{$13$}
    }
};
\end{tikzpicture}
\end{center}

\begin{solution}[1in]
\begin{enumerate}
\item Not balanced
\item Not balanced
\item Balanced
\end{enumerate}

Intuitively, if we can ``stuff'' all the elements in the lowest level into a
higher level, then we can reduce the height of the tree and determine that the
height of the tree is not the same as the height of the optimal binary tree.
\end{solution}
\end{blocksection}
