\question Determine whether an integer is a multiple of 4 using only \texttt{&} and equality operators.

\ifprintanswers\else

\begin{solution}
\begin{lstlisting}
static int isMultipleOfFour(int x) {
    return !(x & 3);
}
\end{lstlisting}
\end{solution}
